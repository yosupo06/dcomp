  • UnionFind

    Declaration

    struct UnionFind;

    UnionFind(Disjoint Set Union)

    Examples

    1. import std.algorithm : equal, sort; auto uf = UnionFind(5); assert(!uf.same(1, 3)); assert(uf.same(0, 0)); uf.merge(3, 2); uf.merge(1, 1); uf.merge(4, 2); uf.merge(4, 3); assert(uf.count == 3); assert(uf.id[2] == uf.id[3]); assert(uf.id[2] == uf.id[4]); assert(equal(uf.group(0), [0])); assert(equal(uf.group(1), [1])); assert(equal(sort(uf.group(2)), [2, 3, 4]));

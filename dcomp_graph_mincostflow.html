dcomp.graph.mincostflow

  • minCostFlow

    Declaration

    MinCostFlowInfo!(C, D, T) minCostFlow(C, D, T)(T g, int s, int t, bool neg);

    最小費用流

    Examples

    1. import std.conv : to; struct Edge { int to, cap, dist, rev; } void addEdge(Edge[][] g, int from, int to, int cap, int dist) { g[from] ~= Edge(to, cap, dist, g[to].length.to!int); g[to] ~= Edge(from, 0, -dist, g[from].length.to!int-1); } auto g = new Edge[][](4); addEdge(g, 0, 1, 10, 3); addEdge(g, 0, 2, 15, 3); addEdge(g, 1, 3, 3, 2); addEdge(g, 2, 3, 20, 4); auto mcfInfo = minCostFlow!(int, int)(g, 0, 3, false); mcfInfo.manyFlow(10^^9); assert(mcfInfo.capFlow == 18); assert(mcfInfo.flow == 3*(3+2) + 15*(3+4));
